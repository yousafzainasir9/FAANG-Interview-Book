Two Pointers (Same Direction)
Remove duplicates, in-place modifications on sorted arrays. One slow pointer for write, one fast for read.
ArrayIn-Place
01
Time
O(n)
Space
O(1)
C#
int left = 0, right = 1;
while (right < nums.Length) {
    if (nums[left] != nums[right]) {
        left++;
        nums[left] = nums[right];
    }
    right++;
}
return left + 1;

Sliding Window (Fixed Size)
Max/min in every k-sized window. Optimized with rolling computation, not re-computing entire window.
ArrayOptimization
03
Time
O(n)
Space
O(k)
C#
int windowSum = 0;
for (int i = 0; i < k; i++) windowSum += nums[i];
int maxSum = windowSum;

for (int i = k; i < nums.Length; i++) {
    windowSum = windowSum - nums[i - k] + nums[i];
    maxSum = Math.Max(maxSum, windowSum);
}
return maxSum;

Binary Search (Exact)

Space
O(1)

    int mid = left + (right - left) / 2;

Union-Find (Disjoint Set)
Efficiently group elements into connected components. Use path compression + union by rank for near O(1) operations.
GraphComponents
09
Time
O(α(n))
Space
O(n)
C#
class UnionFind {
    int[] parent, rank;
    public UnionFind(int n) {
        parent = new int[n];
        rank = new int[n];
        for (int i = 0; i < n; i++) parent[i] = i;
    }

    public int Find(int x) {
        if (parent[x] != x) parent[x] = Find(parent[x]);
        return parent[x];
    }

    public bool Union(int x, int y) {
        int px = Find(x), py = Find(y);
        if (px == py) return false;
        if (rank[px] < rank[py]) (px, py) = (py, px);
        parent[py] = px;
        if (rank[px] == rank[py]) rank[px]++;
        return true;
    }
}

Topological Sort (Kahn)
Order tasks with dependencies. Use indegree-based BFS (Kahn). Detect cycles if result count ≠ n.
DAGDependencies
10
Time
O(V+E)
Space
O(V)
C#
int[] indegree = new int[n];
List[] adj = new List[n];
for (int i = 0; i < n; i++) adj[i] = new();

foreach (int[] edge in edges) {
    adj[edge[0]].Add(edge[1]);
    indegree[edge[1]]++;

Queue q = new();
for (int i = 0; i < n; i++) {
    if (indegree[i] == 0) q.Enqueue(i);
}

List result = new();
while (q.Count > 0) {
    int u = q.Dequeue();
    result.Add(u);
    foreach (int v in adj[u]) {
        indegree[v]--;
        if (indegree[v] == 0) q.Enqueue(v);
    }
}
return result.Count == n ? result : new();

Two Heaps for Median
Find running median in a stream. Max-heap for smaller half, min-heap for larger half. Balance sizes.
HeapStream
12
Time
O(log n)
Space
O(n)
C#
class MedianFinder {
    PriorityQueue maxHeap = new();
    PriorityQueue minHeap = new();

    public void AddNum(int num) {
        maxHeap.Enqueue(num, -num);
        if (maxHeap.Count > minHeap.Count + 1) {
            int val = maxHeap.Dequeue();
            minHeap.Enqueue(val, val);
        }
        if (minHeap.Count > 0 && maxHeap.Peek() > minHeap.Peek()) {
            int max = maxHeap.Dequeue();
            int min = minHeap.Dequeue();
            maxHeap.Enqueue(min, -min);
            minHeap.Enqueue(max, max);
        }
    }

    public double FindMedian() {
        if (maxHeap.Count > minHeap.Count) return maxHeap.Peek();
        return (maxHeap.Peek() + minHeap.Peek()) / 2.0;
    }
}

Backtracking
Generate permutations, combinations, subsets. Use DFS with recursive calls. Prune early when needed.
RecursionExplore
16
Time
O(2^n)
Space
O(n)
C#
void Backtrack(List current, int[] nums,
                  HashSet used, List> result) {
    if (current.Count == nums.Length) {
        result.Add(new List(current));
        return;
    }

    for (int i = 0; i < nums.Length; i++) {
        if (!used.Contains(i)) {
            current.Add(nums[i]);
            used.Add(i);

            Backtrack(current, nums, used, result);

            current.RemoveAt(current.Count - 1);
            used.Remove(i);
        }
    }
}

Trie (Prefix Tree)
Fast word search and prefix matching. O(m) to search m-length word, independent of dictionary size.
StringTrie
17
Insert
O(m)
Space
O(ALPHABET_SIZE × N)
C#
class TrieNode {
    Dictionary children = new();
    bool isEnd = false;
}

class Trie {
    TrieNode root = new();

    public void Insert(string word) {
        TrieNode node = root;
        foreach (char c in word) {
            if (!node.children.ContainsKey(c)) {
                node.children[c] = new TrieNode();
            }
            node = node.children[c];
        }
        node.isEnd = true;
    }

    public bool Search(string word) {
        TrieNode node = root;
        foreach (char c in word) {
            if (!node.children.ContainsKey(c)) return false;
            node = node.children[c];
        }
        return node.isEnd;
    }
}

KMP (String Matching)
Find substring in O(n+m) time using failure function. Better than naive O(n×m).
StringMatching
18
Time
O(n+m)
Space
O(m)
C#
int[] BuildFailure(string pattern) {
    int m = pattern.Length;
    int[] fail = new int[m];
    for (int i = 1; i < m; i++) {
        int j = fail[i - 1];
        while (j > 0 && pattern[i] != pattern[j]) {
            j = fail[j - 1];
        }
        if (pattern[i] == pattern[j]) j++;
        fail[i] = j;
    }
    return fail;
}

int KMPSearch(string text, string pattern) {
    int[] fail = BuildFailure(pattern);
    int j = 0;
    for (int i = 0; i < text.Length; i++) {
        while (j > 0 && text[i] != pattern[j]) {
            j = fail[j - 1];
        }
        if (text[i] == pattern[j]) j++;
        if (j == pattern.Length) return i - j + 1;
    }
    return -1;
}

Monotonic Stack
Find next/previous greater/smaller element in O(n). Stack keeps elements in order, pop smaller when adding larger.
StackMonotonic
21
Time
O(n)
Space
O(n)
C#
// Next Greater Element

Stack stack = new();

for (int i = nums.Length - 1; i >= 0; i--) {
    while (stack.Count > 0 && nums[stack.Peek()] <= nums[i]) {
        stack.Pop();
    }
    if (stack.Count > 0) {

Fast & Slow Pointers
Detect linked list cycles, find middle, reverse. Slow moves 1 step, fast moves 2 steps each iteration.

O(n)
Space
O(1)

// Detect cycle
ListNode slow = head, fast = head;
while (fast != null && fast.next != null) {
    slow = slow.next;
    fast = fast.next.next;
    if (slow == fast) return true; // Cycle found
}
return false;

// Find middle
slow = head; fast = head;
while (fast != null && fast.next != null) {
    slow = slow.next;
    fast = fast.next.next;
}
return slow; // slow is at middle

TIP 01
Integer Overflow
C# ints are 32-bit (-2^31 to 2^31-1). Use long for arithmetic to avoid overflow: int mid = left + (right - left) / 2; instead of (left + right) / 2.
TIP 02
Null Checks
Always check for null before dereferencing: if (node != null && node.val > 0). C# has nullable reference types; turn them on for safety.
TIP 03
Array Bounds
Array indices are 0 to length-1. Check i < nums.Length before accessing nums[i]. Off-by-one errors are common.
TIP 04
String Immutability
Strings are immutable in C#. Use StringBuilder for concatenation in loops: new StringBuilder().Append(...).ToString().
TIP 05
Dictionary Access
Use GetValueOrDefault() or check ContainsKey() to avoid KeyNotFoundException: int count = dict.GetValueOrDefault(key, 0);.
