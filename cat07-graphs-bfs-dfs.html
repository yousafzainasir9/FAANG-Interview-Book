Graph Representations in C#

Two primary ways to represent graphs: adjacency lists and adjacency matrices. Lists are memory-efficient for sparse graphs; matrices work better for dense graphs.

csharp 
// Adjacency List (sparse graphs) — most common
Dictionary<int, List<int>> graph = new();
// Access: graph[node] gives neighbors

// Adjacency Matrix (dense graphs)
int[][] matrix = new int[n][];
// matrix[i][j] = weight or presence

// Weighted edges with node class
public class Node {
    public int Val { get; set; }
    public List<Node> Neighbors { get; set; } = new();
}

BFS Template (Adjacency List)

csharp 
public void BFS(int start, Dictionary<int, List<int>> graph) {
    var visited = new HashSet<int>();
    var queue = new Queue<int>();

    queue.Enqueue(start);
    visited.Add(start);

    while (queue.Count > 0) {
        int node = queue.Dequeue();
        // Process node

        foreach (var neighbor in graph[node]) {
            if (!visited.Contains(neighbor)) {
                visited.Add(neighbor);
                queue.Enqueue(neighbor);
            }
        }
    }
}

DFS Template (Recursive)

csharp 
private void DFS(int node, Dictionary<int, List<int>> graph, HashSet<int> visited) {
    visited.Add(node);
    // Process node

    foreach (var neighbor in graph[node]) {
        if (!visited.Contains(neighbor)) {
            DFS(neighbor, graph, visited);
        }
    }
}

Union-Find (Disjoint Set Union)

csharp 
public class UnionFind {
    private int[] parent, rank;

    public UnionFind(int n) {
        parent = new int[n];
        rank = new int[n];
        for (int i = 0; i < n; i++) parent[i] = i;
    }

    public int Find(int x) {
        if (parent[x] != x)
            parent[x] = Find(parent[x]); // Path compression
        return parent[x];
    }

    public bool Union(int x, int y) {
        int px = Find(x), py = Find(y);
        if (px == py) return false;
        // Union by rank
        if (rank[px] < rank[py]) {
            parent[px] = py;
        } else if (rank[px] > rank[py]) {
            parent[py] = px;
        } else {
            parent[py] = px;
            rank[px]++;
        }
        return true;
    }
}

Topological Sort (Kahn's Algorithm)

csharp 
public List<int> TopologicalSort(Dictionary<int, List<int>> graph, int n) {
    var indegree = new int[n];
    foreach (var neighbors in graph.Values) {
        foreach (var neighbor in neighbors) {
            indegree[neighbor]++;
        }
    }

    var queue = new Queue<int>();
    for (int i = 0; i < n; i++) {
        if (indegree[i] == 0) queue.Enqueue(i);
    }

    var result = new List<int>();
    while (queue.Count > 0) {
        int node = queue.Dequeue();
        result.Add(node);
        foreach (var neighbor in graph[node]) {
            if (--indegree[neighbor] == 0) queue.Enqueue(neighbor);
        }
    }
    return result;
}

Number of Islands

Count distinct islands in a grid. Adjacent 1s (horizontal/vertical) form one island.

Medium Google Amazon Meta
1

Problem Statement

Given an m x n 2D binary grid representing a map of '1's (land) and '0's (water), return the number of islands. An island is surrounded by water and formed by connecting adjacent lands horizontally or vertically.

Approach

1
Iterate through each cell in the grid
2
When finding a '1' not yet visited, increment island count and mark all connected land via BFS or DFS
3
Return total islands found

Solution

csharp 
public int NumIslands(char[][] grid) {
    if (grid == null || grid.Length == 0) return 0;

    int rows = grid.Length, cols = grid[0].Length;
    int islands = 0;

    for (int i = 0; i < rows; i++) {
        for (int j = 0; j < cols; j++) {
            if (grid[i][j] == '1') {
                DFS(grid, i, j, rows, cols);
                islands++;
            }
        }
    }
    return islands;
}

private void DFS(char[][] grid, int r, int c, int rows, int cols) {
    if (r < 0 || r >= rows || c < 0 || c >= cols || grid[r][c] == '0') return;

    grid[r][c] = '0'; // Mark as visited
    DFS(grid, r + 1, c, rows, cols);
    DFS(grid, r - 1, c, rows, cols);
    DFS(grid, r, c + 1, rows, cols);
    DFS(grid, r, c - 1, rows, cols);
}
Time
O(m × n)
Space
O(m × n)
💡
FAANG Tip: Mention you mark cells as visited by changing them in-place. Ask if modifying input is acceptable; if not, use a separate visited set or boolean array.
Clone Graph

Create a deep copy of an undirected graph. Return a reference to the cloned graph.

Medium Google LinkedIn Apple
2

Problem Statement

Given a reference to a node in a connected undirected graph, return a deep copy (clone) of the graph. You must return the copy of the given node as a reference to the cloned graph.

Approach

Use BFS or DFS with a HashMap to track original nodes to cloned nodes. For each unvisited node, create its clone and recursively clone all neighbors.

Solution

csharp 
public class Node {
    public int Val { get; set; }
    public List<Node> Neighbors { get; set; }
}

public Node CloneGraph(Node node) {
    if (node == null) return null;

    var clones = new Dictionary<Node, Node>();
    var queue = new Queue<Node>();

    queue.Enqueue(node);
    clones[node] = new Node { Val = node.Val, Neighbors = new List<Node>() };

    while (queue.Count > 0) {
        var curr = queue.Dequeue();
        foreach (var neighbor in curr.Neighbors) {
            if (!clones.ContainsKey(neighbor)) {
                clones[neighbor] = new Node { Val = neighbor.Val, Neighbors = new List<Node>() };
                queue.Enqueue(neighbor);
            }
            clones[curr].Neighbors.Add(clones[neighbor]);
        }
    }
    return clones[node];
}
Time
O(N + E)
Space
O(N)
⚠️
Edge Case: Cycles in graph—use visited map to avoid infinite loops. Watch for null input and self-loops.
Pacific Atlantic Water Flow

Find cells from which water flows to both Pacific and Atlantic oceans.

Medium Google Meta
3

Problem Statement

Given an m x n matrix of heights, find all cells from which water flows to both the Pacific and Atlantic oceans. Water flows from higher to lower elevations.

Approach

Run multi-source DFS from ocean borders. Start from Pacific border (top and left edges) and Atlantic border (bottom and right edges). A cell reaches the answer if it's reachable from both oceans.

Solution

csharp 
public List<List<int>> PacificAtlantic(int[][] heights) {
    int m = heights.Length, n = heights[0].Length;
    bool[][] pacific = new bool[m][], atlantic = new bool[m][];
    for (int i = 0; i < m; i++) {
        pacific[i] = new bool[n];
        atlantic[i] = new bool[n];
    }

    // DFS from Pacific borders
    for (int i = 0; i < m; i++) {
        DFS(heights, i, 0, m, n, pacific, int.MinValue);
        DFS(heights, i, n - 1, m, n, atlantic, int.MinValue);
    }
    for (int j = 0; j < n; j++) {
        DFS(heights, 0, j, m, n, pacific, int.MinValue);
        DFS(heights, m - 1, j, m, n, atlantic, int.MinValue);
    }

    var result = new List<List<int>>();
    for (int i = 0; i < m; i++) {
        for (int j = 0; j < n; j++) {
            if (pacific[i][j] && atlantic[i][j]) {
                result.Add(new List<int> { i, j });
            }
        }
    }
    return result;
}

private void DFS(int[][] heights, int r, int c, int m, int n, bool[][] ocean, int prevH) {
    if (r < 0 || r >= m || c < 0 || c >= n || ocean[r][c] || heights[r][c] < prevH) return;

    ocean[r][c] = true;
    DFS(heights, r + 1, c, m, n, ocean, heights[r][c]);
    DFS(heights, r - 1, c, m, n, ocean, heights[r][c]);
    DFS(heights, r, c + 1, m, n, ocean, heights[r][c]);
    DFS(heights, r, c - 1, m, n, ocean, heights[r][c]);
}
Time
O(m × n)
Space
O(m × n)
💡
FAANG Tip: Explain "reverse thinking"—instead of checking if water flows to oceans, start FROM the oceans and flood inward. Water can flow to ocean if current height ≥ previous.
Course Schedule

Detect cycle in course dependencies via topological sort.

Medium Amazon Uber
4

Problem Statement

Given n courses (0 to n-1) and a list of prerequisite pairs, determine if you can complete all courses. Course [a,b] means you must complete b before a.

Approach

Use Kahn's algorithm (topological sort with indegrees). If we can sort all n courses, no cycle exists.

Solution

csharp 
public bool CanFinish(int numCourses, int[][] prerequisites) {
    var graph = new List<int>[numCourses];
    var indegree = new int[numCourses];

    for (int i = 0; i < numCourses; i++) graph[i] = new List<int>();

    foreach (var prereq in prerequisites) {
        graph[prereq[1]].Add(prereq[0]);
        indegree[prereq[0]]++;
    }

    var queue = new Queue<int>();
    for (int i = 0; i < numCourses; i++) {
        if (indegree[i] == 0) queue.Enqueue(i);
    }

    int count = 0;
    while (queue.Count > 0) {
        int course = queue.Dequeue();
        count++;
        foreach (var next in graph[course]) {
            if (--indegree[next] == 0) queue.Enqueue(next);
        }
    }

    return count == numCourses;
}
Time
O(V + E)
Space
O(V + E)
ℹ️
Alternative: DFS with colors (white/gray/black) also detects cycles. Black = fully explored, gray = in progress. Gray-to-gray edge = cycle.
Course Schedule II

Return ordering of courses (topological sort order).

Medium Google Amazon
5

Problem Statement

Return an ordering of courses such that each course is taken after all its prerequisites. Return empty array if impossible.

Solution

csharp 
public int[] FindOrder(int numCourses, int[][] prerequisites) {
    var graph = new List<int>[numCourses];
    var indegree = new int[numCourses];

    for (int i = 0; i < numCourses; i++) graph[i] = new List<int>();

    foreach (var prereq in prerequisites) {
        graph[prereq[1]].Add(prereq[0]);
        indegree[prereq[0]]++;
    }

    var queue = new Queue<int>();
    for (int i = 0; i < numCourses; i++) {
        if (indegree[i] == 0) queue.Enqueue(i);
    }

    var result = new int[numCourses];
    int idx = 0;

    while (queue.Count > 0) {
        int course = queue.Dequeue();
        result[idx++] = course;

        foreach (var next in graph[course]) {
            if (--indegree[next] == 0) queue.Enqueue(next);
        }
    }

    return idx == numCourses ? result : new int[0];
}

Graph Valid Tree

Verify if n nodes with n-1 edges form a valid tree using Union-Find.

Medium LinkedIn
6

Problem Statement

A valid tree has exactly n nodes and n-1 edges, and is fully connected with no cycles.

Approach

Use Union-Find. For each edge, union two nodes. If both already belong to the same component (union returns false), a cycle exists. If all edges are processed and we have exactly 1 component, it's a valid tree.

Solution

csharp 
public bool ValidTree(int n, int[][] edges) {
    if (edges.Length != n - 1) return false;

    var uf = new UnionFind(n);

    foreach (var edge in edges) {
        if (!uf.Union(edge[0], edge[1])) return false; // Cycle detected
    }

    return true;
}
Time
O(E·α(N))
Space
O(N)
💡
Tree Properties: n nodes, n-1 edges, connected, acyclic. Quick validation: n - 1 edges is essential.
Number of Connected Components

Count connected components in an undirected graph using Union-Find.

Medium Google
7

Problem Statement

Given n nodes (0 to n-1) and a list of edges, count the number of connected components in the graph.

Solution

csharp 
public int CountComponents(int n, int[][] edges) {
    var uf = new UnionFind(n);

    foreach (var edge in edges) {
        uf.Union(edge[0], edge[1]);
    }

    var roots = new HashSet<int>();
    for (int i = 0; i < n; i++) {
        roots.Add(uf.Find(i));
    }

    return roots.Count;
}

Word Ladder

Find shortest path from beginWord to endWord using BFS.

Hard Google Amazon
8

Problem Statement

Given two words (beginWord, endWord) and a dictionary, find the shortest path from beginWord to endWord. Each step changes exactly one letter. Return the length of the path.

Approach

BFS explores all neighbors at each level. For each word, generate all one-letter-change variants and check if they exist in the dictionary.

Solution

csharp 
public int LadderLength(string beginWord, string endWord, List<string> wordList) {
    var dict = new HashSet<string>(wordList);
    if (!dict.Contains(endWord)) return 0;

    var queue = new Queue<string>();
    var visited = new HashSet<string>();
    queue.Enqueue(beginWord);
    visited.Add(beginWord);

    int level = 1;

    while (queue.Count > 0) {
        int size = queue.Count;
        for (int i = 0; i < size; i++) {
            var word = queue.Dequeue();
            if (word == endWord) return level;

            var neighbors = GetNeighbors(word, dict);
            foreach (var neighbor in neighbors) {
                if (!visited.Contains(neighbor)) {
                    visited.Add(neighbor);
                    queue.Enqueue(neighbor);
                }
            }
        }
        level++;
    }

    return 0;
}

private List<string> GetNeighbors(string word, HashSet<string> dict) {
    var neighbors = new List<string>();
    var chars = word.ToCharArray();

    for (int i = 0; i < chars.Length; i++) {
        char original = chars[i];
        for (char c = 'a'; c <= 'z'; c++) {
            chars[i] = c;
            var newWord = new string(chars);
            if (dict.Contains(newWord)) neighbors.Add(newWord);
        }
        chars[i] = original;
    }

    return neighbors;
}
Time
O(N·L·26)
Space
O(N)
💡
Optimization: Bidirectional BFS halves the search space. Start from both beginWord and endWord simultaneously.
Alien Dictionary

Derive alien alphabet order from sorted word list using topological sort.

Hard Google Amazon
9

Problem Statement

Given a sorted list of words in an alien language, derive the order of the alphabet. Return empty string if invalid.

Approach

Compare adjacent words character-by-character. When first differing character is found, it defines an order relationship. Build a graph and perform topological sort.

Solution

csharp 
public string AlienOrder(List<string> words) {
    var graph = new Dictionary<char, List<char>>();
    var indegree = new Dictionary<char, int>();

    // Initialize all characters
    foreach (var word in words) {
        foreach (char c in word) {
            if (!graph.ContainsKey(c)) {
                graph[c] = new List<char>();
                indegree[c] = 0;
            }
        }
    }

    // Build graph from word pairs
    for (int i = 0; i < words.Count - 1; i++) {
        var w1 = words[i], w2 = words[i + 1];
        int minLen = Math.Min(w1.Length, w2.Length);

        for (int j = 0; j < minLen; j++) {
            if (w1[j] != w2[j]) {
                if (!graph[w1[j]].Contains(w2[j])) {
                    graph[w1[j]].Add(w2[j]);
                    indegree[w2[j]]++;
                }
                break;
            }
        }
    }

    // Topological sort
    var queue = new Queue<char>();
    foreach (var c in indegree.Keys) {
        if (indegree[c] == 0) queue.Enqueue(c);
    }

    var sb = new StringBuilder();
    while (queue.Count > 0) {
        char c = queue.Dequeue();
        sb.Append(c);

        foreach (var neighbor in graph[c]) {
            if (--indegree[neighbor] == 0) queue.Enqueue(neighbor);
        }
    }

    return sb.Length == graph.Count ? sb.ToString() : "";
}
Time
O(N·L + U)
Space
O(U)
⚠️
Edge Case: Invalid input (e.g., "abc" before "ab"—longer comes before prefix). Return empty string if graph size ≠ topological sort size.
Surrounded Regions

Flip surrounded 'O' regions to 'X' via border DFS.

Medium Meta Microsoft
10

Problem Statement

Given a 2D board, capture all 'O' surrounded by 'X'. A surrounded 'O' region has no path to board edges.

Approach

DFS from all border 'O' cells to mark unreachable 'O's. Then flip all remaining 'O' to 'X'.

Solution

csharp 
public void Solve(char[][] board) {
    int m = board.Length, n = board[0].Length;

    // Mark all border-reachable 'O' with '#'
    for (int i = 0; i < m; i++) {
        DFS(board, i, 0, m, n);
        DFS(board, i, n - 1, m, n);
    }
    for (int j = 0; j < n; j++) {
        DFS(board, 0, j, m, n);
        DFS(board, m - 1, j, m, n);
    }

    // Flip: unsurrounded 'O' → 'O', surrounded 'O' → 'X'
    for (int i = 0; i < m; i++) {
        for (int j = 0; j < n; j++) {
            if (board[i][j] == '#') board[i][j] = 'O';
            else if (board[i][j] == 'O') board[i][j] = 'X';
        }
    }
}

private void DFS(char[][] board, int r, int c, int m, int n) {
    if (r < 0 || r >= m || c < 0 || c >= n || board[r][c] != 'O') return;

    board[r][c] = '#';
    DFS(board, r + 1, c, m, n);
    DFS(board, r - 1, c, m, n);
    DFS(board, r, c + 1, m, n);
    DFS(board, r, c - 1, m, n);
}
Time
O(m × n)
Space
O(m × n)
💡
Key Insight: Only 'O' regions NOT touching borders are captured. Start from borders and mark all reachable 'O's as safe.
Rotting Oranges

Find time for all fresh oranges to rot via multi-source BFS.

Medium Google Amazon Meta
11

Problem Statement

Given a grid with fresh (1), rotten (2), and empty (0) oranges, rotten oranges spread to adjacent fresh oranges each minute. Return minutes until all fresh oranges rot, or -1 if impossible.

Approach

Multi-source BFS: enqueue all rotten oranges initially. Process level-by-level (each level = 1 minute). Count minutes and fresh oranges remaining.

Solution

csharp 
public int OrangesRotting(int[][] grid) {
    int m = grid.Length, n = grid[0].Length;
    int freshCount = 0;
    var queue = new Queue<(int, int)>();

    for (int i = 0; i < m; i++) {
        for (int j = 0; j < n; j++) {
            if (grid[i][j] == 2) queue.Enqueue((i, j));
            else if (grid[i][j] == 1) freshCount++;
        }
    }

    int[][] dirs = new int[][] { new int[] { 1, 0 }, new int[] { -1, 0 }, new int[] { 0, 1 }, new int[] { 0, -1 } };
    int minutes = 0;

    while (queue.Count > 0 && freshCount > 0) {
        int size = queue.Count;
        for (int i = 0; i < size; i++) {
            var (r, c) = queue.Dequeue();
            foreach (var dir in dirs) {
                int nr = r + dir[0], nc = c + dir[1];
                if (nr >= 0 && nr < m && nc >= 0 && nc < n && grid[nr][nc] == 1) {
                    grid[nr][nc] = 2;
                    queue.Enqueue((nr, nc));
                    freshCount--;
                }
            }
        }
        minutes++;
    }

    return freshCount == 0 ? minutes : -1;
}
Time
O(m × n)
Space
O(m × n)
ℹ️
Multi-source BFS: Start with multiple sources in queue. Excellent for spreading/propagation problems where multiple origins expand simultaneously.
Shortest Path in Binary Matrix

Find shortest path from top-left to bottom-right allowing 8-directional movement.

Medium Google
12

Problem Statement

Given an n×n binary matrix, find shortest path from (0,0) to (n-1,n-1). You can move in 8 directions; 0 = walkable, 1 = obstacle. Return -1 if no path.

Solution

csharp 
public int ShortestPathBinaryMatrix(int[][] grid) {
    int n = grid.Length;
    if (grid[0][0] == 1) return -1;

    var queue = new Queue<(int, int)>();
    queue.Enqueue((0, 0));
    grid[0][0] = 1; // Mark as visited

    int[][] dirs = new int[][] {
        new int[] { 0, 1 }, new int[] { 1, 0 }, new int[] { 0, -1 }, new int[] { -1, 0 },
        new int[] { 1, 1 }, new int[] { 1, -1 }, new int[] { -1, 1 }, new int[] { -1, -1 }
    };

    int steps = 1;

    while (queue.Count > 0) {
        int size = queue.Count;
        for (int i = 0; i < size; i++) {
            var (r, c) = queue.Dequeue();
            if (r == n - 1 && c == n - 1) return steps;

            foreach (var dir in dirs) {
                int nr = r + dir[0], nc = c + dir[1];
                if (nr >= 0 && nr < n && nc >= 0 && nc < n && grid[nr][nc] == 0) {
                    grid[nr][nc] = 1;
                    queue.Enqueue((nr, nc));
                }
            }
        }
        steps++;
    }

    return -1;
}
Time
O(N²)
Space
O(N²)
Network Delay Time

Calculate minimum time for signal to reach all nodes via Dijkstra's algorithm.

Medium Google Amazon
13

Problem Statement

Given n nodes and directed weighted edges (travel time), return the time when all nodes receive a signal sent from node 1.

Approach

Dijkstra's algorithm: maintain shortest distance to each node. Always process the nearest unvisited node. Return max distance.

Solution

csharp 
public int NetworkDelayTime(int[][] times, int n, int k) {
    var graph = new Dictionary<int, List<(int, int)>>();
    for (int i = 1; i <= n; i++) graph[i] = new List<(int, int)>();

    foreach (var time in times) {
        graph[time[0]].Add((time[1], time[2]));
    }

    var dist = new int[n + 1];
    for (int i = 0; i <= n; i++) dist[i] = int.MaxValue;
    dist[k] = 0;

    var pq = new PriorityQueue<int, int>();
    pq.Enqueue(k, 0);

    while (pq.Count > 0) {
        int node = pq.Dequeue();

        foreach (var (neighbor, time) in graph[node]) {
            if (dist[node] + time < dist[neighbor]) {
                dist[neighbor] = dist[node] + time;
                pq.Enqueue(neighbor, dist[neighbor]);
            }
        }
    }

    int maxTime = 0;
    for (int i = 1; i <= n; i++) {
        if (dist[i] == int.MaxValue) return -1;
        maxTime = Math.Max(maxTime, dist[i]);
    }

    return maxTime;
}
Time
O((V+E)logV)
Space
O(V + E)
💡
Dijkstra's vs Bellman-Ford: Dijkstra faster but requires non-negative weights. Bellman-Ford handles negative weights but slower.
Cheapest Flights Within K Stops

Find minimum cost route with ≤K stops using Bellman-Ford or BFS with distance tracking.

Medium Google Uber
14

Problem Statement

Given n cities, flights with prices, and a source/destination, find minimum cost to travel with at most k stops (edges).

Solution (Bellman-Ford)

csharp 
public int FindCheapestPrice(int n, int[][] flights, int src, int dst, int k) {
    var dist = new int[n];
    for (int i = 0; i < n; i++) dist[i] = int.MaxValue;
    dist[src] = 0;

    // Relax edges k times (k stops = k+1 edges)
    for (int i = 0; i < k + 1; i++) {
        var temp = new int[n];
        for (int j = 0; j < n; j++) temp[j] = dist[j];

        foreach (var flight in flights) {
            int from = flight[0], to = flight[1], price = flight[2];
            if (dist[from] != int.MaxValue && dist[from] + price < temp[to]) {
                temp[to] = dist[from] + price;
            }
        }
        dist = temp;
    }

    return dist[dst] == int.MaxValue ? -1 : dist[dst];
}
Time
O(K × E)
Space
O(N)
⚠️
K Stops vs K Edges: k stops = k edges. Relax k+1 times. Without stopping at source, relax k times for exactly k edges.
Reconstruct Itinerary

Find Eulerian path through directed multigraph (flight tickets).

Hard Google Amazon
15

Problem Statement

Given a list of tickets [from, to], reconstruct your travel itinerary starting from "JFK". Each ticket must be used exactly once.

Approach

Hierholzer's algorithm: DFS to find Eulerian path in a directed multigraph. Use recursion and backtracking.

Solution

csharp 
public List<string> FindItinerary(List<List<string>> tickets) {
    var graph = new Dictionary<string, List<string>>();

    // Build graph and sort destinations
    foreach (var ticket in tickets) {
        string from = ticket[0], to = ticket[1];
        if (!graph.ContainsKey(from)) graph[from] = new List<string>();
        graph[from].Add(to);
    }

    foreach (var key in graph.Keys) {
        graph[key].Sort();
    }

    var result = new List<string>();
    DFS("JFK", graph, result);
    result.Reverse();

    return result;
}

private void DFS(string from, Dictionary<string, List<string>> graph, List<string> result) {
    if (graph.ContainsKey(from)) {
        while (graph[from].Count > 0) {
            string to = graph[from][0];
            graph[from].RemoveAt(0);
            DFS(to, graph, result);
        }
    }
    result.Add(from);
}
